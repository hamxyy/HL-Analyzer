namespace SHS.SAT.FittingHIDBInterfaces.HearingLanguageRuntime {
using System;
using System.Diagnostics;
using System.Collections.Generic;
using Identifiers;
using HIPlatformData.D8.HearingAidLanguage;
namespace shs_sat {
public class BasicLibrary_MPO : MacroBase {
public int MPO_Type_Get( ) {
if ( _hi[HIControls.d8SwMpoType] == HIPositions.d8SwMpoType_Both ) {
Debug.WriteLine("[HAL]BLIB:MPO_Type_Get = Multichannel-both" );
return 1;
}
if ( _hi[HIControls.d8SwMpoType] == HIPositions.d8SwMpoType_Broadband ) {
Debug.WriteLine("[HAL]BLIB:MPO_Type_Get = Broadband" );
return 0;
}
else {
Debug.WriteLine("[HAL]BLIB:MPO_Type_Get = ERROR = channel alone!!" );
return 1;
}
}
public void MPO_Type_Set( int pos ) {
if ( pos == 1 ) {
_hi[HIControls.d8SwMpoType] = HIPositions.d8SwMpoType_Both;
Debug.WriteLine("[HAL]BLIB:MPO_Type_Set = Multichannel" );
}
else {
_hi[HIControls.d8SwMpoType] = HIPositions.d8SwMpoType_Broadband;
Debug.WriteLine("[HAL]BLIB:MPO_Type_Set = Broadband" );
}
}
public void MPO_Slider_Set( int totalSlider, int sliderX ) {
do {
if ( _hi[HIControls.d8SwChanOutLim][sliderX]< HIPositions.d8SwChanOutLim_M12dB ) {
Debug.WriteLine("[HAL]BLIB-MPO_Slider_Set is above < -12dB for slider " + sliderX );
_hi[HIControls.d8SwChanOutLim][sliderX] = HIPositions.d8SwChanOutLim_M12dB;
}
++sliderX;
}
while( sliderX <= totalSlider );
}
}
}
}
