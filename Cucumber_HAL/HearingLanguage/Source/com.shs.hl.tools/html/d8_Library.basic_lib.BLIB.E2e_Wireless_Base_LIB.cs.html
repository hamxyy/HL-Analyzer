namespace SHS.SAT.FittingHIDBInterfaces.HearingLanguageRuntime {
using System;
using System.Diagnostics;
using System.Collections.Generic;
using Identifiers;
using HIPlatformData.D8.HearingAidLanguage;
namespace shs_sat {
public class BasicLibrary_E2e_Wireless : MacroBase {
public bool E2e_WirelessEn_Get( ) {
bool retval = false;
if ( _hi[HIControls.d8PhlWlsOn] == HIPositions.d8PhlWlsOn_on ) {
retval = true;
Debug.WriteLine("[HAL]:BLIB.E2e_WirelessEn_Get = true " );
}
else {
retval = false;
Debug.WriteLine("[HAL]:BLIB.E2e_WirelessEn_Get = false " );
}
return retval;
}
public void E2e_WirelessEn_Set( int pos ) {
if ( pos == 40 ) {
_hi[HIControls.d8PhlWlsOn] = HIPositions.d8PhlWlsOn_onff;
}
else if ( pos == 30 ) {
_hi[HIControls.d8PhlWlsOn] = HIPositions.d8PhlWlsOn_e2eOff_WlpOn;
}
else if ( pos == 20 ) {
_hi[HIControls.d8PhlWlsOn] = HIPositions.d8PhlWlsOn_e2eOff_WlpOff;
}
else {
_hi[HIControls.d8PhlWlsOn] = HIPositions.d8PhlWlsOn_off;
}
Debug.WriteLine("[HAL]:BLIB.E2e_WirelessEn_Set = " + pos );
}
}
}
}
