using System.Diagnostics;
using System.Collections.Generic;
using Identifiers;
using HIPlatformData.D8.HearingAidLanguage;
namespace shs_sat {
public class High_Level_Library_AutoPhone : MacroBase {
public void AutoPhoneEn_Set( int pos ) {
int env_pgm = _env[EnvParameters.envProg];
int Previous_AutophoneEnabled_HP = ((shs_sat.BasicLibrary_AutoPhone)GetMacro<shs_sat.BasicLibrary_AutoPhone>()).AutoPhoneEn_Get_HPx( );
if ( pos == 1 ) {
if ( ((shs_sat.Mid_Level_Library_AutoPhone)GetMacro<shs_sat.Mid_Level_Library_AutoPhone>()).AutoPhoneEn_IsVisible( ) ) {
Debug.WriteLine("[HAL]HLIB.AutophoneEn_Set to ON for" + env_pgm );
if ( Previous_AutophoneEnabled_HP >= 0 && Previous_AutophoneEnabled_HP < 6 ) {
((shs_sat.BasicLibrary_AutoPhone)GetMacro<shs_sat.BasicLibrary_AutoPhone>()).AutoPhoneEn_Set_HPx_OFF( Previous_AutophoneEnabled_HP );
((shs_sat.BasicLibrary_AccessibleManually)GetMacro<shs_sat.BasicLibrary_AccessibleManually>()).AccessibleManuallyEn_Set_HPx( Previous_AutophoneEnabled_HP , 1 );
}
((shs_sat.BasicLibrary_AutoPhone)GetMacro<shs_sat.BasicLibrary_AutoPhone>()).AutoPhoneEn_Set_HPx_ON( _env[EnvParameters.envProg]);
}
}
else {
((shs_sat.BasicLibrary_AutoPhone)GetMacro<shs_sat.BasicLibrary_AutoPhone>()).AutoPhoneEn_Set_HPx_OFF( _env[EnvParameters.envProg]);
((shs_sat.BasicLibrary_AccessibleManually)GetMacro<shs_sat.BasicLibrary_AccessibleManually>()).AccessibleManuallyEn_Set_HPx( _env[EnvParameters.envProg], 1 );
}
}
public int AutoPhoneEn_Get( ) {
int AutophoneEnabled_HP = ((shs_sat.BasicLibrary_AutoPhone)GetMacro<shs_sat.BasicLibrary_AutoPhone>()).AutoPhoneEn_Get_HPx( );
if ( AutophoneEnabled_HP == _env[EnvParameters.envProg]) {
return 1;
}
else {
return 0;
}
}
}
